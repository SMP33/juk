111 NavigationNode::calculateVelocity(double abs_speed, GeoMath::v3 offset, GeoMath::v3 current_velocity,uint8_t ctrl_mode)
112 {
113  ctrl_flag = 5;
114  const double max_break_acc = 3;
115  const double max_force_acc = 0.7;
116  double need_abs_speed ;
117 
118  double max_z_speed = 5;
119  double current_speed = current_velocity.length_xyz();
120 
121  need_abs_speed = std::min(abs_speed, current_speed + max_force_acc);
122  double addition_break_time = 0.1;
123  double current_distance = offset.length_xyz();
124 
125  double break_distance = 0;
126 
127  switch (ctrl_mode)
128  {
129  case juk_msg::juk_set_target_data_msg::mode_allow_break_distance:
130 
131  break_distance = ((abs_speed*abs_speed) / (2*max_break_acc));
132 
133  if (current_distance < break_distance + addition_break_time*need_abs_speed+1.5)
134  {
135  ctrl_flag = 7;
136  }
137  break;
138  }
139 
140  GeoMath::v2 cC(cos(position_data.course*GeoMath::CONST.DEG2RAD), sin(position_data.course*GeoMath::CONST.DEG2RAD));
141  GeoMath::v2 cN(cos(target.course*GeoMath::CONST.DEG2RAD), sin(target.course*GeoMath::CONST.DEG2RAD));
142 
143  yaw_rate = -cC.angle_xy(cN)*GeoMath::CONST.RAD2DEG/2;
144 
145  velocity_need = offset.normalize_xyz(need_abs_speed);
146  //std::cout.flags(std::ios::fixed);
147  //std::cout << offset << std::endl;
148  if (abs(yaw_rate) < 3)
149  {
150  if (velocity_need.z > max_z_speed)
151  velocity_need = velocity_need * (max_z_speed / velocity_need.z);
152 
153  if (ctrl_flag == 7)
154  {
155  velocity_need.x = offset.x;
156  velocity_need.y = offset.y;
157  velocity_need.z = offset.z/2;
158  }
159  }
160  else
161  {
162  ctrl_flag = 5;
163  velocity_need.x = 0;
164  velocity_need.y = 0;
165  velocity_need.z = 0;
166  }
